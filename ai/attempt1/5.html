Пиздец с кодом — C++

// C++ пример пиздец кода, чтоб светился и искрил
#include <iostream>
using namespace std;

int main() {
  cout << "Литуизм - ебучий ад на твоём процессоре!" << endl;
  for(int i=0; i<5; i++){
    cout << "Хуй в твоём коде! #" << i+1 << endl;
  }
  return 0;
}
