Литуевский Мат и Туториал Ассемблера от Белки за 1 минуту

Туториал по Ассемблеру от Белки за 1 минуту

; Быстрый пример кода на x86 NASM
section .data
    msg db "Привет, мир от Белки!", 0

section .text
    global _start

_start:
    ; write(1, msg, 20)
    mov eax, 4
    mov ebx, 1
    mov ecx, msg
    mov edx, 20
    int 0x80

    ; exit(0)
    mov eax, 1
    xor ebx, ebx
    int 0x80

Этот код выводит строку "Привет, мир от Белки!" в консоль Linux.

Чтобы скомпилировать:

  • nasm -f elf32 file.asm -o file.o
  • ld -m elf_i386 file.o -o file
  • ./file
